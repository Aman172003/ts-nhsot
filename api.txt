
-Get all users

curl -X GET \  -H "Content-Type: application/json" \
  http://localhost:3000/users

--------------------------------------------------------------------------------------------------------

-Create a new user
--replace <userId> with any one user id from users list

curl -X POST \
  -H "Content-Type: application/json" \
  -d '{
    "name": "arpit@gmail.com",
    "password": "123456789",
    "locale": "en",
    "displayName": "Arpit"
  }' \
  http://localhost:3000/users

--------------------------------------------------------------------------------------------------------

-Delete a user
--replace <id> with id of any users from users list

curl -X DELETE \
  -H "Content-Type: application/json" \
  http://localhost:3000/users/<id>

--------------------------------------------------------------------------------------------------------

-Get all todo list

curl -X GET \  -H "Content-Type: application/json" \
  http://localhost:3000/todos

--------------------------------------------------------------------------------------------------------

-Create a new todo
--replace <userId> with any one user id from users list

curl -X POST \
  -H "Content-Type: application/json" \
  -d '{
    "name": "Study",
    "isCompleted": false,
    "userId": "<userId>"
  }' \
  http://localhost:3000/todos

--------------------------------------------------------------------------------------------------------

-Update the todo activity
--replace <id> with any one id from todo list
--replace <name> with any name you want to change to

curl -X PUT \
  -H "Content-Type: application/json" \
  -d '{
    "name": "<name>",
    "isCompleted": false
  }' \
  http://localhost:3000/todos/<id>

--------------------------------------------------------------------------------------------------------

-Delete the todo activity
--replace <id> with any id you want to delete

curl -X DELETE \
  -H "Content-Type: application/json" \
  http://localhost:3000/todos/<id>
